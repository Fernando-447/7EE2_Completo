-- Crear la base de datos
CREATE DATABASE ProyectoV3;

-- Seleccionar la base de datos
USE ProyectoV3;

-- Crear la tabla Alumno
CREATE TABLE Alumno (
    numConAlu VARCHAR(10) NOT NULL,
    apePatAlu VARCHAR(20) NOT NULL DEFAULT 'Sin apellido Paterno',
    apeMatAlu VARCHAR(20) NOT NULL,
    nomAlu VARCHAR(20) NOT NULL,
    
    CONSTRAINT pk_cveAlu PRIMARY KEY (numConAlu)
);

-- Crear la tabla Docente
CREATE TABLE Docente (alumno
    numTraDoc SMALLINT NOT NULL,
    nom VARCHAR(50) NOT NULL,
    apePat VARCHAR(50) NOT NULL DEFAULT 'Sin apellido Paterno',
    apeMat VARCHAR(50) NOT NULL DEFAULT 'Sin apellido Materno',
    
    CONSTRAINT pk_cveDoc PRIMARY KEY (numTraDoc)
    
);

CREATE TABLE MaterialNoConsumible (
    cveMatNoCon INT NOT NULL,
    codUTNG VARCHAR(15) NOT NULL UNIQUE,
    desMatNoCon VARCHAR(100) DEFAULT 'Sin descripcion',
    estMatNoCon ENUM('Roto', 'Desgastado', 'Regular') NOT NULL DEFAULT 'Regular',
    disposicion ENUM('En stock', 'Prestada') NOT NULL DEFAULT 'En stock',


    CONSTRAINT pk_cveMatNoCon PRIMARY KEY (cveMatNoCon)
);


CREATE TABLE MaterialConsumible (
    cveMatCon INT NOT NULL,
    desMatCon VARCHAR(100) DEFAULT 'Sin descripcion',
    estMatCon ENUM('Roto', 'Desgastado', 'Regular') NOT NULL DEFAULT 'Regular',
    disposicion ENUM('En stock', 'Prestada') NOT NULL DEFAULT 'En stock',

    CONSTRAINT pk_cveMatCon PRIMARY KEY (cveMatCon)
);
